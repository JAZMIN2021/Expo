button_down = False

# Inicializa una variable para mantener un seguimiento del estado
showing_image = False

def show_menu(images):
    """ Shows the menu on the screen"""
    
    # bring in the global variables
    global line, highlight, shift, total_images

    # menu variables
    item = 1
    line = 1
    line_height = 10

    # clear the display
    oled.fill_rect(0, 0, width, height, 0)

    # Shift the list of images so that it shows on the display
    short_list = images[shift:shift+total_lines]

    for index, item in enumerate(short_list, start=1):
        if highlight == line:
            oled.fill_rect(0, (line - 1) * line_height, width, line_height, 1)
            oled.text(">", 0, (line - 1) * line_height)
            oled.text("Image " + str(index), 10, (line - 1) * line_height)
            oled.show()
        else:
            oled.text("Image " + str(index), 10, (line - 1) * line_height)
            oled.show()
        line += 1

def launch_image(image):
    """ Display the selected image """
    # clear the screen
    oled.fill_rect(0, 0, width, height, 0)
    oled.blit(image, 0, 0)
    oled.show()

# Display the menu

# Repeat forever
while True:
    if up_button.value() == False and not button_down:
        button_down = True
        if highlight > 1:
            highlight -= 1
        else:
            if shift > 0:
                shift -= 1
        show_menu(image_data)

    if down_button.value() == False and not button_down:
        button_down = True
        if highlight < total_lines:
            highlight += 1
        else:
            if shift + total_lines < total_images:
                shift += 1
        show_menu(image_data)

    if select_button.value() == False and not button_down:
        button_down = True
        if showing_image:
            # Si ya se está mostrando una imagen, regresa al menú
            showing_image = False
            show_menu(image_data)
        else:
            # Muestra la imagen seleccionada
            print("Launching Image", highlight - 1 + shift)
            launch_image(image_fb[(highlight - 1) + shift])
            showing_image = True

    # Debounce buttons
    if up_button.value() == True and down_button.value() == True and select_button.value() == True:
        button_down = False
